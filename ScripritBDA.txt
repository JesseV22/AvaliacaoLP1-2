CREATE TABLE aluno (
id INT PRIMARY KEY,
nome_al VARCHAR(50),
email_al VARCHAR(50),
idade_al INT,
curso_id INT,
FOREIGN KEY (curso_id) REFERENCES curso(id)
);

CREATE TABLE curso (
id INT PRIMARY KEY,
nome_cur VARCHAR(50),
descricao_cur VARCHAR(200)
);

CREATE TABLE matricula (
    id INT PRIMARY KEY,
    aluno_mar INT,
    curso_mar INT,
    data_mar DATE,
    FOREIGN KEY (aluno_mar) REFERENCES aluno(id),
    FOREIGN KEY (curso_mar) REFERENCES curso(id)
);
