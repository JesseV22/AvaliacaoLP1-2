CREATE TABLE curso (
    id INT PRIMARY KEY,
    nome_cur VARCHAR(50),
    descri_cur VARCHAR(200)
    carga_cur INT

);


CREATE TABLE aluno (
    id INT PRIMARY KEY,
    nome_al VARCHAR(50),
    email_al VARCHAR(50),
    idade_al INT,
    tel_al VARCHAR(50),
    
);


CREATE TABLE professor (
    id INT PRIMARY KEY,
    nome_prof VARCHAR(50),
    email_prof VARCHAR(50),
    end_prof VARCHAR(100),
    tel_prof VARCHAR(20),   
    especialidade VARCHAR(50),
    
    
    
);


CREATE TABLE matricula (
    id INT PRIMARY KEY,
    ID INT PRIMARY KEY,
    AlunoID INT,
    CursoID INT,
    DataMatricula DATE,
    FOREIGN KEY (AlunoID) REFERENCES Aluno(ID),
    FOREIGN KEY (CursoID) REFERENCES Curso(ID)
);
